% ------------------------------------- Esercizio 1
\addpoints
\question
Determina le soluzioni delle seguenti espressioni di secondo grado:\\
\begin{parts}
\part[5]
\(\dfrac{4}{3}x^2 - \dfrac{1}{4} = 0\)\\
\fillwithdottedlines{0.5in}
{\footnotesize
\begin{solution}
	\(x_1 = \dfrac{\sqrt{3}}{4}; \, x_2 = -\dfrac{\sqrt{3}}{4}\)
\end{solution}
}
\vspace{.75cm}

\part[5]
\(x^2 + 8x + 12 = 0\)\\
\fillwithdottedlines{0.75in}
{\footnotesize
\begin{solution}
	\(x_1 = -2; \, x_2 = -6 \)
\end{solution}
}
\end{parts}
\vspace{.5cm}

% ------------------------------------- Esercizio 2

\fillwithdottedlines{0.75in}

{\footnotesize
\begin{solution}
	\(x_1 = -2 \quad x_2=\dfrac{1}{2}\)
