\question
determina le {\em condizioni di esistenza} e, ove possibile, semplifica le seguenti frazioni algebriche utilizzando i metodi di fattorizzazione;\\
\begin{parts}
\part[2]
 \(\dfrac{a^2 + 2ab + b^2}{a^2 - b^2}\)
% \fillwithlines{.5in}
{\footnotesize
\begin{solution}

	\(\dfrac{a + b}{a - b}\)\\
	
	\(C.E. = \{a \neq b\}\)
\end{solution}
}
\vspace{0.5cm}

\part[3] 
\(\dfrac{4x^3 + 8x^2 + 4x}{4x^2 - 4x}\)
%\fillwithlines{.5in}
{\footnotesize
\begin{solution}

	\(\dfrac{x + 1}{x - 1}\)\\
	
	\(C.E. = \{x \neq 1\}\)
\end{solution}
}
\vspace{0.5cm}
\part[3] 
\(\dfrac{t^5 - 5t^4 + t - 5}{t^2 - 25}\)
% \fillwithlines{.5in}
{\footnotesize
\begin{solution}

	\(\dfrac{t^4 + 1}{t + 5}\)\\
	
	\(C.E. = \{t \neq -5\}\)
\end{solution}
}
\vspace{.5cm}
\end{parts}
